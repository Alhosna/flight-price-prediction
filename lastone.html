Feature Engineering

#Process Date and Weekday Features

df$date <- as.Date(df$date, format = "%d-%m-%Y")
df$weekday <- wday(df$date, week_start = 1) # Monday = 1, Sunday = 7
df$is_weekend <- as.factor(ifelse(df$weekday %in% c(6, 7), "Weekend", "Weekday"))
df$weekday_sin <- sin(2 * pi * df$weekday / 7)
df$weekday_cos <- cos(2 * pi * df$weekday / 7)
df <- df %>% select(-date, -weekday)
df$time_of_day <- as.factor(case_when(
  df$arr_time < 6 ~ "Night",
  df$arr_time < 12 ~ "Morning",
  df$arr_time < 18 ~ "Afternoon",
  TRUE ~ "Evening"
))
# Convert Flight Duration (e.g., "02h 15m") to numeric (hours)
df$time_taken <- sapply(df$time_taken, function(x) {
  if (is.na(x)) return(NA)
  hours <- as.numeric(str_extract(x, "\\d+(?=h)"))
  minutes <- as.numeric(str_extract(x, "\\d+(?=m)"))
  if (is.na(minutes)) minutes <- 0
  hours + minutes / 60
})
# Clean Stop Column and Drop Unneeded Columns
 df$stop <- str_squish(df$stop)
 df$stop <- ifelse(grepl("1-stop", df$stop), "1-stop", as.character(df$stop))
df$stop <- as.factor(df$stop)
df$num_code <- NULL
df$ch_code <- NULL
